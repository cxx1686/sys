CREATE TABLE `work_order` (  `id` int(10) unsigned NOT NULL AUTO_INCREMENT,  `order_id` int(10) unsigned NOT NULL COMMENT '订单号',  `member_id` smallint(6) NOT NULL COMMENT '负责人',  `wo_type` varchar(250) NOT NULL COMMENT '工单类型',  `img` varchar(100) NOT NULL COMMENT '图片',  `remarks` text NOT NULL COMMENT '备注',  `add_time` int(10) NOT NULL COMMENT '创建时间',  `create_member_id` smallint(6) NOT NULL COMMENT '创建人',  `status` tinyint(1) unsigned NOT NULL DEFAULT '1' COMMENT '处理状态1未处理2已处理',  PRIMARY KEY (`id`),  KEY `order_id` (`order_id`)) ENGINE=MyISAM AUTO_INCREMENT=2 DEFAULT CHARSET=utf8
